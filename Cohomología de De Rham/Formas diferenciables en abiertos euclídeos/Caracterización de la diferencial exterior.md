0\longrightarrow \Omega^{0}(U) \xrightarrow{\hspace{4pt}d\hspace{4pt}}\Omega^{1}(U) \longrightarrow \cdots \longrightarrow  \Omega^{p}(U) \xrightarrow{\hspace{4pt}d\hspace{4pt}}\Omega^{p+1}(U) \longrightarrow \cdots
$$
